\begin{homeworkProblem}[Item 1: Week 3 - Linear Search and Duplicate Finder]{}

\homeworkSection{ 1. Pseudocode for linear search}

This Simple Algorithm demonstrates how to perform a linear search. \\
\textbf{Input:} This algorithm takes a populated array $A$ and a value to search for $v$, as parameters. \\
\textbf{Output:} The Algorithm is a boolean type and returns either True or False respective of whether the $v$ was found in the list or not.


\begin{algorithm}
\begin{algorithmic}[1]
\caption{LinearSearch}
\Procedure{BOOL LINEARSEARCH}{$v$, $A$[\hspace{1mm}]}
\For {each element $i$ in $A$}
\If{ $A[i] = v$}\\
\indent \indent \indent	\Return true
\EndIf
\EndFor\\
\Return false
\EndProcedure
\end{algorithmic}
\end{algorithm}



\homeworkSection{ 2. Pseudocode for finding duplicates in a list}

This algorithm demonstrates how to examine if a list has duplicate entries using a linear search.\\
\textbf{Input:} This algorithm takes a populated array $A$ as a parameter.\\
\textbf{Output:} This Algorithm is a boolean type and returns true or false respective of whether a duplicate value is found or not.
\begin{algorithm}
\caption{Examining for duplicates}
\begin{algorithmic}[1]
\Procedure{BOOL EXFORDUPES}{$A$[\hspace{1mm}]}
\For { each element $i$ in $A$[\hspace{1mm}]}
\For { each element $j$ in $A$[\hspace{1mm}]}
\If{ $A[i] = A[j]$}\\
\indent \indent \indent \indent \Return true
\EndIf
\EndFor
\EndFor
\EndProcedure
\end{algorithmic}
\end{algorithm}
\end{homeworkProblem}

\homeworkSection{Additional work: Critical values of relative runtimes}
Write a function that determines the critical value at which the relative runtime of two linear algorithms swap.\\ 
For this algorithm, I am assuming that $k1$ \textgreater $k2$ ( Expression 1 \textgreater Expression 2, when $ n=0$).\\
The Algorithm is very simple; While the value of Expression 1 is greater than Expression 2, increase the value of $n$. \\When the Runtime of the algorithms swap, the while-loop exit condition is fulfilled and the current value of n is returned.

\begin{algorithm}
\caption{Relative runtime comparison algorithm}
\begin{algorithmic}[1]
\Procedure{CRITVAL}{$m1, k1, m2, k2$}
 
\While{$(m1 * n + k1)\textgreater (m2*n+k2)$} 
\State $n++$
\EndWhile \\
\Return n\\
\EndProcedure
\end{algorithmic}
\end{algorithm}

\begin{homeworkProblem}[Item 3: Week 6 - Harmonic Series]{}

\homeworkSection{1. Harmonic Series (Pseudocode)}
Use pseudocode to specify a recursive algorithm to compute the nth value of the harmonic series, for some integer n.\\

The Harmonic series is as follows: $ 1+1/2+1/3+1/4+1/5+..1/n$\\
\textbf{Input:} This algorithm takes two parameters $t$ and $n$ which are the total sum of the algorithm and the number of repetitions, respectively.\\
\textbf{Output:} This algorithm outputs the value $t$ which is the total sum of the harmonic series. \\

This procedure uses a while-loop to control the number recursive iterations.\\
While the number of iterations left is above 0, add the next value to $t$, decrease the number of iterations by 1 and recursively call the procedure with the new values of $t$ and $n$.\\When the number of iterations left is no longer above 0, the final value of $t$ is returned and the procedure ends.
\begin{algorithm}
\caption{Computing nth value of harmonic series}
\begin{algorithmic}[1]
\Procedure{HARM}{float $t$, float $n$}
\While{ $n>0$}
\State $t \gets t + (1/n)$
\State $n--$
\State $HARM$($t,n$)
\EndWhile\\
\Return t
\EndProcedure
\end{algorithmic}
\end{algorithm}
